## MLP 网络（向量模式）

```python
class MLPNet(nn.Module):
    """14 维 → 128 → 128 → 5 动作"""
    def __init__(self, obs_dim=14, n_actions=5, hidden=128):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(obs_dim, hidden), nn.ReLU(),
            nn.Linear(hidden, hidden), nn.ReLU(),
        )
        self.actor = nn.Linear(hidden, n_actions)
        self.critic = nn.Linear(hidden, 1)

        # 权重初始化: 正交初始化, gain=0.5 防初期输出过大
        for m in self.modules():
            if isinstance(m, nn.Linear):
                nn.init.orthogonal_(m.weight, gain=0.5)
                nn.init.constant_(m.bias, 0)
```

参数数量: 14×128 + 128 + 128×128 + 128 + 128×5 + 5 + 128×1 + 1 ≈ **19K**

## CNN 网络（像素模式）

```python
class CNNNet(nn.Module):
    """Nature DQN 风格 CNN, 160×160×3 → 5 动作"""
    def __init__(self, in_ch=3, n_actions=5):
        super().__init__()
        self.conv = nn.Sequential(
            nn.Conv2d(in_ch, 32, 8, 4), nn.ReLU(),   # 160→39
            nn.Conv2d(32, 64, 4, 2), nn.ReLU(),       # 39→18
            nn.Conv2d(64, 64, 3, 1), nn.ReLU(),       # 18→16
            nn.Flatten(),
        )
        # 自动计算 FC 输入维度
        with torch.no_grad():
            dummy = torch.zeros(1, in_ch, 160, 160)
            fc_in = self.conv(dummy).shape[1]          # 64×16×16 = 16384

        self.fc = nn.Sequential(
            nn.Linear(fc_in, 256), nn.ReLU(),
        )
        self.actor = nn.Linear(256, n_actions)
        self.critic = nn.Linear(256, 1)
```

参数数量: ~500K（大部分在 FC 层 16384×256）

## PPO Trainer

### 核心方法

```python
class PPOTrainer:
    def __init__(self, net, lr=1e-3):
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.net = net.to(self.device)
        self.opt = torch.optim.Adam(self.net.parameters(), lr=lr)
        self.buf = {"s": [], "a": [], "lp": [], "r": [], "v": [], "d": []}

    def select_action(self, obs, eval_mode=False):
        t = torch.FloatTensor(obs).unsqueeze(0).to(self.device)
        with torch.no_grad():
            a, lp, ent, v = self.net.act(t, det=eval_mode)
        return a.item(), lp.item(), ent.item(), v.item()

    def store(self, obs, a, lp, r, v, d):
        """存入经验缓冲区"""
        self.buf["s"].append(obs); self.buf["a"].append(a)
        self.buf["lp"].append(lp); self.buf["r"].append(r)
        self.buf["v"].append(v); self.buf["d"].append(d)

    def update(self, gamma=0.99, lam=0.95, clip=0.2, epochs=4, batch=128,
               ent_coef=0.02, critic_coef=0.5):
        """PPO 策略更新 + GAE 优势估计"""
```

### GAE（广义优势估计）
```python
adv = np.zeros_like(R)
gae = 0.0
for t in range(len(R) - 1, -1, -1):
    next_v = 0.0 if t == len(R) - 1 else V[t + 1]
    delta = R[t] + gamma * next_v * (1 - D[t]) - V[t]
    gae = delta + gamma * lam * (1 - D[t]) * gae
    adv[t] = gae
ret = adv + V  # 价值目标 = 优势 + 状态价值

# 标准化优势（稳定训练）
adv = (adv - adv.mean()) / (adv.std() + 1e-8)
```

### PPO Clip Loss
```python
ratio = torch.exp(new_log_prob - old_log_prob)
surr1 = ratio * advantage
surr2 = torch.clamp(ratio, 1-clip, 1+clip) * advantage
actor_loss = -torch.min(surr1, surr2).mean()

critic_loss = F.mse_loss(new_value, returns)
entropy_bonus = -entropy.mean()

total_loss = actor_loss + 0.5*critic_loss - 0.02*entropy_bonus
```

### Clip 机制解释
`surr2 = clamp(ratio, 0.8, 1.2) * advantage`

当 advantage > 0（好动作）:
- ratio > 1.2 时被 clip 到 1.2，不会无限增大
- 防止策略一次更新得太激进

当 advantage < 0（坏动作）:
- ratio < 0.8 时被 clip 到 0.8
- 防止过度惩罚导致策略崩溃

## 训练循环

```python
def train(visual=False, episodes=2000, render=False, log_path=None):
    mode = "visual" if visual else "vector"
    env = SimpleRunner(mode=mode)

    if visual:
        net = CNNNet(n_actions=5)
    else:
        net = MLPNet(obs_dim=14, n_actions=5)

    agent = PPOTrainer(net, lr=1e-3)

    for ep in range(1, episodes + 1):
        obs, _ = env.reset()
        done = False
        ep_r = 0.0

        while not done:
            a, lp, ent, v = agent.select_action(obs)
            next_obs, r, done, trunc, _ = env.step(a)
            agent.store(obs, a, lp, r, v, done)
            obs = next_obs
            ep_r += r
            if trunc: break

        # 每局结束记录统计
        r100.append(ep_r)
        avg_r = np.mean(r100)

        # PPO 更新（缓冲区够大时才更新）
        loss = agent.update()

        # 日志
        if ep % 25 == 0:
            print(f"{ep:5d} {env.steps:6d} {env.score:7.0f} "
                  f"{ep_r:8.1f} {avg_r:8.1f} {elapsed:6.0f}s")

        # 最佳模型保存
        if avg_r > best_avg and len(r100) >= 50:
            best_avg = avg_r
            agent.save(f"models/simple_{label}_best.pth")
```

def play_ai(model_path):

    # 从 key 名自动判断: 包含 "conv" → CNN

        net = CNNNet(n_actions=5)
